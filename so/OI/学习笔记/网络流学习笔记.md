## 最大流（Dinic）

```cpp
const int N = 2e6 + 10, inf = 0x3f3f3f3f;
struct E {
  int next, to, w;
};
struct D {
  int tot = -1, head[N], cur[N];
  E e[N];
  inline void init() { memset(head, -1, sizeof(head)); }
  inline void add(int x, int y, int z) {
    e[++tot] = (E){head[x], y, z}, head[x] = tot;
    e[++tot] = (E){head[y], x, 0}, head[y] = tot;
  }

  int hd, tl, q[N], dis[N];
  inline bool bfs() {
    memset(dis, -1, sizeof(dis));
    hd = tl = 0, q[tl] = s;
    dis[s] = 0, cur[s] = head[s];
    while (hd <= tl) {
      int u = q[hd++];
      for (int i = head[u], v; ~i; i = e[i].next) {
        if (dis[v = e[i].to] == -1 && e[i].w) {
          dis[v] = dis[u] + 1, cur[v] = head[v];
          if (v == t) return 1;
          q[++tl] = v;
        }
      }
    }
    return 0;
  }
  inline int dfs(int u, int lim) {
    if (u == t) return lim;
    int flow = 0, delta;
    for (int i = cur[u], v; ~i && flow < lim; i = e[i].next) {
      cur[u] = i;
      if (dis[v = e[i].to] == dis[u] + 1 && e[i].w) {
        delta = dfs(v, min(e[i].w, lim - flow));
        if (!delta) dis[v] = -1;
        e[i].w -= delta, e[i ^ 1].w += delta;
        flow += delta;
      }
    }
    return flow;
  }
  inline int Dinic() {
    int tmp = 0;
    while (bfs()) tmp += dfs(s, inf);
    return tmp;
  }
} T;
```

## 最小费用最大流（SPFA+EK）

```cpp
const int N = 2e6 + 10, inf = 0x3f3f3f3f;
struct E {
  int next, to, val, cos;
};
struct EK {
  int tot = 1, s, t, head[N], pre[N];
  int dis[N];
  bool vis[N];
  E e[N];
  inline void add(int x, int y, int val, int cos) {
    e[++tot] = (E){head[x], y, val, cos}, head[x] = tot;
    e[++tot] = (E){head[y], x, 0, -cos}, head[y] = tot;
  }
  inline bool SPFA() {
    memset(vis, 0, sizeof(vis));
    memset(pre, 0, sizeof(pre));
    for (int i = 1; i <= t; ++i) dis[i] = inf;
    deque<int> q;
    q.push_back(s);
    dis[s] = 0;
    while (!q.empty()) {
      int u = q.front();
      q.pop_front();
      for (int i = head[u], v; i; i = e[i].next) {
        if (dis[v = e[i].to] > dis[u] + e[i].cos && e[i].val) {
          dis[v] = dis[u] + e[i].cos, pre[v] = i;
          if (!vis[v]) {
            vis[v] = 1;
            if (!q.empty() && dis[v] < dis[q.front()]) {
              q.push_front(v);
            } else {
              q.push_back(v);
            }
          }
        }
      }
      vis[u] = 0;
    }
    return dis[t] != inf;
  }
  int maxFlow, sumCost;
  inline void getCost() {
    int flow = 0x7fffffff;
    for (int i = t; i != s; i = e[pre[i] ^ 1].to)
      flow = min(flow, e[pre[i]].val);
    maxFlow += flow;
    for (int i = t; i != s; i = e[pre[i] ^ 1].to) {
      e[pre[i]].val -= flow, e[pre[i] ^ 1].val += flow;
      sumCost += flow * e[pre[i]].cos;
    }
  }
  inline void solve() {
    while (SPFA()) getCost();
  }
}
```

## 餐巾计划问题

[P1251 餐巾计划问题](https://www.luogu.com.cn/problem/P1251)

### 题目描述

一个餐厅在相继的 $N$ 天里,每天需用的餐巾数不尽相同。假设第 $i$ 天需要 $r_i$块餐巾( i=1,2,...,N)。餐厅可以购买新的餐巾,每块餐巾的费用为 $p$ 分;或者把旧餐巾送到快洗部,洗一块需 m 天,其费用为 f 分;或者送到慢洗部,洗一块需 $n$ 天($n>m$),其费用为 $s$ 分($s<f$)。

每天结束时,餐厅必须决定将多少块脏的餐巾送到快洗部,多少块餐巾送到慢洗部,以及多少块保存起来延期送洗。但是每天洗好的餐巾和购买的新餐巾数之和,要满足当天的需求量。

试设计一个算法为餐厅合理地安排好 $N$ 天中餐巾使用计划,使总的花费最小。编程找出一个最佳餐巾使用计划。

### 输入输出格式

#### 输入格式

由标准输入提供输入数据。文件第 1 行有 1 个正整数 $N$，代表要安排餐巾使用计划的天数。

接下来的一行是餐厅在相继的 $N$ 天里,每天需用的餐巾数。

最后一行包含5个正整数$p,m,f,n,s$。$p$ 是每块新餐巾的费用; $m$ 是快洗部洗一块餐巾需用天数; $f$ 是快洗部洗一块餐巾需要的费用; $n$ 是慢洗部洗一块餐巾需用天数; $s$ 是慢洗部洗一块餐巾需要的费用。

#### 输出格式

将餐厅在相继的 N 天里使用餐巾的最小总花费输出

### 输入输出样例

#### 输入样例 #1

```
3
1 7 5 
11 2 2 3 1
```

#### 输出样例 #1

```
134
```

### 说明

ri<=10000000

### 题解

发现早上要处理干净餐巾，晚上要处理脏餐巾，所以考虑拆点，为什么要拆点呢？因为早晚要处理不同的数据，而且复杂程度（需要处理的数据规模）不太一样，若是只留着一个点是没法进行很多操作的。

为方便叙述，以二元组$(val, cos)$表示一条边，边权为$val$，费用为$cos$；源点为$s$，汇点为$t$；白天记为$i\ (i\in[1,n])$，晚上记为$i+n$。

我们只要让白天有足够多的干净餐巾就可以，怎么来的可以先不用管，所以白天$i$就向$t$连一条$(r_i,0)$，再从$s$到$i$连边$(r_i,0)$，保证要有$r_i$的流量。

源点向白天连的边$(r_i,0)$其实没有必要，因为到$t$的流量最多只有$r_i$，这个限制条件解决当前问题（白天要有足够餐巾）完全足够，况且后面要从$s$到$i$连边$(inf,p)$。若没有这个$(inf,p)$的话从$s$到$i$连一条$(inf,0)$足以使最大流量是$r_i$。

现在考虑餐巾的来头。

- 可以直接购买，无需其他夜晚的继承。故从$s$到$i$连边$(inf,p)$，也就是想买多少买多少，但是每块要花$p$这么多钱。

- 可以是洗过的餐巾。所以我们直接考虑在夜晚发生了什么，因为白天只是得到餐巾，并未涉及洗脏餐巾的问题。从夜晚 $i+n$ 到 $i+m\ (i+m\in[1,n])$ 连 $(inf,w)$，其中 $m$ 代表快洗或慢洗对应的天数，$w$ 对应其费用。也就是说，可以从这个晚上随便拿多少餐巾去洗，在 $i+m$ 这天的白天收到干净的餐巾。
- 可以是延迟送洗洗过的。我们不知道它延迟了多少块或者多少天，我们只关心它延迟了，所以从 $i+n$ 到 $i+n+1$ 连一条 $(inf,0)$，也就是当天晚上可以从前一天晚上拿到随便多少脏餐巾，而且不花钱。

而且要限制夜晚得到的餐巾量，就从 $s$ 向 $i+n$ 连一条 $(r_i,0)$ 就好了，也就是在夜晚这里的流只能最多从源点（其实相当于白天）获得 $r_i$ 的流量（从之前夜晚获得的不算）。

因此这五部分连边构成了要跑的这个网络，跑最小费用最大流即可。

我再解释一下为什么要跑费用流qwq

因为我们要在限制流量 $r_i$（即一天获得的餐巾数）的同时还要涉及到各个方案的费用，况且题中说了是最小总花费（虽然题中信息不一定涉及正解）。

我再解释一下为什么能跑最小费用最大流QAQ

我们限制这个图，让它最多只能跑$\sum r_i$的流，而且每一天必须是$r_i$，也就是最大流恰好就满足了题中的要求。在最大流的情况下，任何一种流的费用都是符合情况的，只是多少的区别，所以我们让费用最小，也就组成了最小费用最大流。

## [CTSC1999]家园 / 星际转移问题

[P2754 [CTSC1999]家园 / 星际转移问题](https://www.luogu.com.cn/problem/P2754)

### 题目描述

由于人类对自然资源的消耗，人们意识到大约在 2300 年之后，地球就不能再居住了。于是在月球上建立了新的绿地，以便在需要时移民。令人意想不到的是，2177 年冬由于未知的原因，地球环境发生了连锁崩溃，人类必须在最短的时间内迁往月球。

现有 $n$ 个太空站位于地球与月球之间，且有 $m$ 艘公共交通太空船在其间来回穿梭。每个太空站可容纳无限多的人，而太空船的容量是有限的，第 $i$ 艘太空船只可容纳 $h_i$ 个人。每艘太空船将周期性地停靠一系列的太空站，例如 $(1,3,4)$ 表示该太空船将周期性地停靠太空站 $134134134\dots$。每一艘太空船从一个太空站驶往任一太空站耗时均为 $1$。人们只能在太空船停靠太空站(或月球、地球)时上、下船。

初始时所有人全在地球上，太空船全在初始站。试设计一个算法，找出让所有人尽快地全部转移到月球上的运输方案。

### 输入输出格式

#### 输入格式

输入的第一行是三个用空格隔开的整数，分别代表太空站个数 $n$，太空船个数 $m$ 和地球上的人数 $k$。 

第 $2$ 到第 $(m + 1)$ 行，每行给出一艘太空船的信息，第 $(i + 1)$ 行的第一个整数 $h_i$ 代表第 $i$ 艘太空船可容纳的人数。随后有一个整数 $r_i$，代表第 $i$ 艘太空船停靠的站点数。之后有 $r_i$ 个整数，依次代表该太空船停靠站点的编号 $S_{i, j}$，其中太空站自 $1$ 至 $n$ 编号，地球编号为 $0$，月球编号为 $-1$。

#### 输出格式

输出一行一个整数，代表将所有人转移到月球上的最短用时。若无解则输出 $0$。

### 输入输出样例

#### 输入样例 #1

```
2 2 1
1 3 0 1 2
1 3 1 2 -1
```

#### 输出样例 #1

```
5
```

### 说明

#### 数据规模与约定

对于 $100\%$ 的数据，保证： 

- $1 \leq n \leq 13$。
- $1 \leq m \leq 20$。
- $1 \leq k \leq 50$。
- $1 \leq r_i \leq n + 2$。
- $-1 \leq S_{i, j}\leq n$。

### 题解

挺经典的一个靠建模的题

 首先判断无解的条件：地月之间无法到达，即不在一个连通块里。那很容易想到做法了吧——并查集！

至于怎么得出答案，可以发现所有太空船一块运动（以自己的周期），每一步都知道船具体在哪，所以相当于动的只是人！而且是单调向下运动！这东西像什么？分层图啊。

类似的思路可以参考<a href="../fen-ceng-tu-zui-duan-lu-xue-xi-bi-ji">分层图最短路</a>

{% asset_image 1.png %}

借用题解区里一个大佬的图，这些道路分别代表这一些意思

- 前面的红点竖直往下走的边：人们待在这一点不动，时间加了一。
- 最后一列的红点竖直向上走的边：其实就相当于结束节点，表示有一部分到这里的人可以出去了。
- 斜着连的边：表示太空船花一时间单位走到了下一个停靠点。

所以人们只要沿着分层图跑就好了。

然而我们发现这只能判断到每一层后能通过多少人，所以我们可以枚举一下分层图的层数，然后判断到底有多少人能通过（跑最大流），如果超过了$k$的话，也就是$k$个人可以通过。

```cpp
#include <bits/stdc++.h>
#define LOCAL
using namespace std;

namespace STDquantum {

template <typename T> inline void read(T &x) {
  char ch;
  int f = 1;
  while (ch = getchar(), !isdigit(ch)) {
    if (ch == '-') f = -1;
  }
  x = (ch ^ 48);
  while (ch = getchar(), isdigit(ch)) { x = x * 10 + (ch ^ 48); }
  x *= f;
}

const int N = 2e6 + 10, inf = 0x3f3f3f3f;
int n, m, k, s, t, ans, r[N], h[N], S[101][101];

struct UF {
  int parent[N];
  inline void UFset() {
    for (int i = 1; i <= n + 2; ++i) parent[i] = -1;
  }
  inline int Find(int x) {
    int s = x, tmp;
    while (parent[s] >= 0) s = parent[s];
    while (s != x) {
      tmp = parent[x];
      parent[x] = s;
      x = tmp;
    }
    return s;
  }
  inline void Union(int x, int y) {
    int nx = Find(x), ny = Find(y);
    int tmp = parent[nx] + parent[ny];
    parent[nx] = parent[nx] > parent[ny] ? ny : tmp;
    parent[ny] = parent[nx] > parent[ny] ? tmp : nx;
  }
} UF;

struct E {
  int next, to, w;
};
struct D {
  int tot = -1, head[N], cur[N];
  E e[N];
  inline void init() { memset(head, -1, sizeof(head)); }
  inline void add(int x, int y, int z) {
    e[++tot] = (E){head[x], y, z}, head[x] = tot;
    e[++tot] = (E){head[y], x, 0}, head[y] = tot;
  }

  int hd, tl, q[N], dis[N];
  inline bool bfs() {
    memset(dis, -1, sizeof(dis));
    hd = tl = 0, q[tl] = s;
    dis[s] = 0, cur[s] = head[s];
    while (hd <= tl) {
      int u = q[hd++];
      for (int i = head[u], v; ~i; i = e[i].next) {
        if (dis[v = e[i].to] == -1 && e[i].w) {
          dis[v] = dis[u] + 1, cur[v] = head[v];
          if (v == t) return 1;
          q[++tl] = v;
        }
      }
    }
    return 0;
  }
  inline int dfs(int u, int lim) {
    if (u == t) return lim;
    int flow = 0, delta;
    for (int i = cur[u], v; ~i && flow < lim; i = e[i].next) {
      cur[u] = i;
      if (dis[v = e[i].to] == dis[u] + 1 && e[i].w) {
        delta = dfs(v, min(e[i].w, lim - flow));
        if (!delta) dis[v] = -1;
        e[i].w -= delta, e[i ^ 1].w += delta;
        flow += delta;
      }
    }
    return flow;
  }
  inline int Dinic() {
    int tmp = 0;
    while (bfs()) tmp += dfs(s, inf);
    return tmp;
  }
} T;

inline void main() {
  read(n), read(m), read(k);
  s = N - 3, t = N - 2;
  UF.UFset(), T.init();
  for (int i = 1; i <= m; ++i) {
    read(h[i]), read(r[i]);
    for (int j = 0, x, y; j < r[i]; ++j) {
      read(S[i][j]);
      if (!S[i][j]) S[i][j] = n + 1;
      if (j) {
        x = S[i][j], y = S[i][j - 1];
        x = UF.Find(~x ? x : n + 2);
        y = UF.Find(~y ? y : n + 2);
        if (x != y) UF.Union(x, y);
      }
    }
  }
  if (UF.Find(n + 1) != UF.Find(n + 2)) {
    return (void)puts("0");
  }
  int tmp = 0;
  for (ans = 1;; ++ans) {
    T.add(s, ans * (n + 1), inf);
    for (int i = 1, x, y; i <= m; ++i) {
      x = (ans - 1) % r[i], y = ans % r[i];
      x = ~S[i][x] ? (ans - 1) * (n + 1) + S[i][x] : t;
      y = ~S[i][y] ? ans * (n + 1) + S[i][y] : t;
      T.add(x, y, h[i]);
    }
    tmp += T.Dinic();
    if (tmp >= k) return (void)printf("%d", ans);
    for (int i = 1; i <= n + 1; ++i) {
```

## 飞行员配对方案问题

[P2756 飞行员配对方案问题](https://www.luogu.com.cn/problem/P2756)

### 题目背景

第二次世界大战期间，英国皇家空军从沦陷国征募了大量外籍飞行员。由皇家空军派出的每一架飞机都需要配备在航行技能和语言上能互相配合的两名飞行员，其中一名是英国飞行员，另一名是外籍飞行员。在众多的飞行员中，每一名外籍飞行员都可以与其他若干名英国飞行员很好地配合。

### 题目描述

一共有 $n$ 个飞行员，其中有 $m$ 个外籍飞行员和 $(n - m)$ 个英国飞行员，**外籍飞行员从 $1$ 到 $m$ 编号**，**英国飞行员从 $m + 1$ 到 $n$ 编号**。 对于给定的外籍飞行员与英国飞行员的配合情况，试设计一个算法找出最佳飞行员配对方案，使皇家空军一次能派出最多的飞机。

### 输入输出格式

#### 输入格式

输入的第一行是用空格隔开的两个正整数，分别代表外籍飞行员的个数 $m$ 和飞行员总数 $n$。

从第二行起到倒数第二行，每行有两个整数 $u, v$，代表外籍飞行员 $u$ 可以和英国飞行员 $v$ 配合。

输入的最后一行保证为 `-1 -1`，代表输入结束。

#### 输出格式

**本题存在 Special Judge**。 请输出能派出最多的飞机数量，并给出一种可行的方案。 输出的第一行是一个整数，代表一次能派出的最多飞机数量，设这个整数是 $k$。

第 $2$ 行到第 $k + 1$ 行，每行输出两个整数 $u, v$，代表在你给出的方案中，外籍飞行员 $u$ 和英国飞行员 $v$ 配合。这 $k$ 行的 $u$ 与 $v$ 应该互不相同。

### 输入输出样例

#### 输入样例 #1

```
5 10
1 7
1 8
2 6
2 9
2 10
3 7
3 8
4 7
4 8
5 10
-1 -1
```

#### 输出样例 #1

```
4
1 7
2 9
3 8
5 10
```

### 说明

**【数据范围与约定】** 

- 对于 $100\%$ 的数据，保证 $1 \leq m \leq n < 100$，$1 \leq u \leq m < v \leq n$，同一组配对关系只会给出一次。 

**【提示】** 

- 请注意输入的第一行先读入 $m$，再读入 $n$。

### 题解

二分图最大匹配~

英籍和外籍飞行员连边，英籍和$s$，外籍和$t$连，边权都是一。

至于输出方案，一个流流过去了，那么这个边的反向边边权肯定不是$0$。输出起始点和终止点就可以。

```cpp
#include <bits/stdc++.h>
using namespace std;
const int inf = 1e9, N = 1e6 + 10;
int n, m, x, y, z, s, t, maxflow, deep[N], sum;
struct E {
  int next, to, dis;
} edge[N];
int num_edge = -1, head[N], cur[N];
queue<int> q;
void add_edge(int from, int to, int dis, bool flag) {
  edge[++num_edge].next = head[from];
  edge[num_edge].to = to;
  if (flag) edge[num_edge].dis = dis;
  head[from] = num_edge;
}
bool bfs(int s, int t) {
  memset(deep, 0x7f, sizeof(deep));
  while (!q.empty()) q.pop();
  for (int i = 1; i <= sum; ++i) { cur[i] = head[i]; }
  deep[s] = 0;
  q.push(s);
  while (!q.empty()) {
    int now = q.front();
    q.pop();
    for (int i = head[now]; i != -1; i = edge[i].next) {
      if (deep[edge[i].to] > inf && edge[i].dis) {
        deep[edge[i].to] = deep[now] + 1;
        q.push(edge[i].to);
      }
    }
  }
  if (deep[t] < inf)
    return true;
  else
    return false;
}
int dfs(int now, int t, int limit) {
  if (!limit || now == t) return limit;
  int flow = 0, f;
  for (int i = cur[now]; i != -1; i = edge[i].next) {
    cur[now] = i;
    if (deep[edge[i].to] == deep[now] + 1 &&
        (f = dfs(edge[i].to, t, min(limit, edge[i].dis)))) {
      flow += f;
      limit -= f;
      edge[i].dis -= f;
      edge[i ^ 1].dis += f;
      if (!limit) break;
    }
  }
  return flow;
}
void Dinic(int s, int t) {
  while (bfs(s, t)) { maxflow += dfs(s, t, inf); }
}
void Match(int s, int t) {
  for (int i = 1; i <= m; ++i) {
    for (int j = head[i]; j != -1; j = edge[j].next) {
      if (!edge[j].dis && edge[j].to < s)
        printf("%d %d\n", i, edge[j].to);
    }
  }
}
int main() {
  memset(head, -1, sizeof(head));
  scanf("%d%d", &m, &n);
  s = n + m + 1, t = sum = s + 1, z = 1;
  while (scanf("%d%d", &x, &y) != EOF) {
    if (-1 == x && y == -1) break;
    add_edge(x, y, z, 1);
    add_edge(y, x, z, 0);
  }
  for (int i = 1; i <= m; ++i)
    add_edge(s, i, z, 1), add_edge(i, s, z, 0);
  for (int i = m + 1; i <= n + m; ++i)
    add_edge(i, t, z, 1), add_edge(t, i, z, 1);
  Dinic(s, t);
  printf("%d\n", maxflow);
  Match(s, t);
  return 0;
}
```
