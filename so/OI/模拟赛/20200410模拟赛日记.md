#include<algorithm>
#include<cstdio>
#include<cmath>
#include<iostream>
#define Maxn 1000000001
using namespace std;
const double Pi=acos(-1);
int n,m;
struct CP
{
  CP (double xx=0,double yy=0){x=xx,y=yy;}
  double x,y;
  CP operator + (CP const &B) const
  {return CP(x+B.x,y+B.y);}
  CP operator - (CP const &B) const
  {return CP(x-B.x,y-B.y);}
  CP operator * (CP const &B) const
  {return CP(x*B.x-y*B.y,x*B.y+y*B.x);}
}f[Maxn<<1],p[Maxn<<1];
int tr[Maxn<<1];
void fft(CP *f,bool flag)
{
  for (int i=0;i<n;i++)
    if (i<tr[i])swap(f[i],f[tr[i]]);
  //枚举区间长度 
  for(int p=2;p<=n;p<<=1){
    int len=p>>1;//待合并的长度
    CP tG(cos(2*Pi/p),sin(2*Pi/p));
    if(!flag)tG.y*=-1;//p次单位根
    for(int k=0;k<n;k+=p){//枚举起始点 
      CP buf(1,0);//遍历一个子问题并合并
      for(int l=k;l<k+len;l++){
        CP tt=buf*f[len+l];
        f[len+l]=f[l]-tt;
        f[l]=f[l]+tt;
        buf=buf*tG;
      }
    }
  }
}
int main()
{
	int uiu=0;

    tr[i]=(tr[i>>1]>>1)|((i&1)?n>>1:0);
  fft(f,1);fft(p,1);//DFT

//npio10.cpp
#include<cstdio>
#include<algorithm>
#include<vector>
#include<stack>
using namespace std;
const int MN=100000;
vector< pair<int,int> > v[MN+5];
vector<int> ans;
stack<int> S;
int dfn[MN+5],low[MN+5],cnt;
void tarjan(int x,int from)
{
	dfn[x]=low[x]=++cnt;
	S.push(x);
	for(int i=0;i<v[x].size();++i)if(v[x][i].second!=from)
	{
		int y=v[x][i].first;
		if(!dfn[y])
		{
			tarjan(y,v[x][i].second);
			low[x]=min(low[x],low[y]);
			if(low[y]>=dfn[x])
			{
				int res=(x+y)%998244353;
				while(S.top()!=y)
				{
					res=(res+S.top())%998244353;
					S.pop();
				}
				S.pop();
				ans.push_back(res);
			}
		}
		else low[x]=min(low[x],dfn[y]);
	}
}
int main()
{
	int n,m,x,y;
	scanf("%d%d",&n,&m);

	{
		scanf("%d%d",&x,&y);

		v[y].push_back(make_pair(x,i));
	}
	for(int i=1;i<=n;++i)if(!dfn[i])tarjan(i,0);
	sort(ans.begin(),ans.end());
	printf("%d\n",ans.size());
	for(int i=0;i<ans.size();++i)printf("%d ",ans[i]);
}
