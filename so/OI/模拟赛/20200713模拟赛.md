# $Sign$

$(sign.cpp/.in/.out)$

$Time\ limit: 1s$

$Memory\ limit:128M$

## $Description$

这是一道签到题。

给出一棵 $N$ 个节点的树，求所有起点为叶节点的**有向**路径，其上每一条边权值和的和。

## $Input$

第一行是一个整数 $N$ 。

接下来 $N-1$ 行，每行三个整数 $$L_i, A_i, B_i$$ ，代表存在一条连接 $$A_i$$ 和 $$B_i$$ 的长度为 $$L_i$$ 的边。

## $Output$

输出一个整数，代表所求的结果。

## $Sample$

|            $Input$            | $Output$ |
| :---------------------------: | :------: |
| $3$<br>$2\ 1\ 2$<br>$1\ 3\ 1$ |   $9$    |

$Tips:$合法的路径有四条路径，长度分别为$1$，$2$，$3$，$3$，和为$9$。

## $Hint$

对于所有的数据，边长 $L_i$ 在 $int$ 范围内。

|  $ID$  |    $N$    |
| :----: | :-------: |
| $1-3$  | $\le300$  |
| $4-7$  | $\le3000$ |
| $8-10$ | $\le10^5$ |

## 样例

### 输入#1

```
3
2 1 2
1 3 1
```

### 输出#1

```
9
```

## 做题

令人大呼毒瘤！[看这](#Result)

具体做法就是枚举边统计贡献

一条边对答案的贡献就是

（左边叶子节点数$\times$右边总节点数$+$左边总节点数$\times$右边叶子结点数）$\times$边权

这个建议手模一下就可以了

然后考虑如何统计，首先上面需要的两个值可以用树形DP维护子树总节点数和子树叶子节点数。

由于枚举的是边，而并不知道哪个节点是要取出状态的节点，如图一，如果红色标记的边代表现在枚举的边的话，那么$x$节点的子树总节点数和子树叶子节点数是不能直接用的，所以考虑如何区分$x$与$y$。显然$y$的深度比$x$深，而$y$节点取出的内容均是可用的。

{% asset_image 2.png %}

然后就考虑如何求外面的叶子节点数和总节点数，总节点数好说，用$n-f[y]$就可以了，而全局的总叶子节点数，可以统计节点的度，为一的就是叶子节点，也可以直接用$d[root]$来代表，那么我们就会发现如果像图二的情况，总叶子节点就会少算一个，那么我们不妨取一个一定不是叶子节点的节点当根。

然后$sum$这里记得多乘几个```1ll```。

```cpp
#include <bits/stdc++.h>
using namespace std;

namespace STDquantum {

const int N = 4e5 + 10;
int n, head[N], tot, f[N], d[N], hei[N], t[N], root;
    /*
    f[N]: 子树总节点数
    d[N]: 子树总叶子节点数
    hei[N]: 节点高度
    t[N]: 节点度数
    root: 选取的根节点
    */
long long sum;
bool vis[N];
struct E {
    int next, from, to, w;
} e[N], r[N];
inline void add(int from, int to, int w) {
    e[++tot].next = head[from];
    e[tot].to = to;
    head[from] = tot;
}
void dfs(int s, int h) {
    vis[s] = 1, hei[s] = h, f[s] = 1;
    // 维护子树总节点数的时候先加上自己，f数组赋成1
    bool xn = 0; // 判断是不是叶子结点
    for(int i = head[s]; i; i = e[i].next) {
        if (!vis[e[i].to]) {
            xn = 1; // 如果有出边就不是叶子结点
            dfs(e[i].to, h + 1);
            f[s] += f[e[i].to];
            d[s] += d[e[i].to];
            // 树形DP维护总子节点数和叶子节点数
        }
    }
    if (!xn) ++d[s]; // 如果是叶子节点就赋成1（之前是0可以直接++）
}
inline void main() {
    scanf("%d", &n);
    for(int i = 1, x, y, z; i < n; ++i) {
        scanf("%d%d%d", &z, &x, &y);
        add(x, y, z), add(y, x, z); // 无向图
        r[i] = (E){0, x, y, z}; // 记录边
        ++t[x], ++t[y]; // 边的尽头节点度数加一
    }
    for(int i = 1; i <= n; ++i) {
        if (t[i] >= 2) { // 选取任意一个度数大于一的节点做根
            root = i;
            break;
        }
    }
    dfs(root, 1);
    for(int i = 1, x, y, z; i < n; ++i) {
        x = r[i].from, y = r[i].to, z = r[i].w;
        if (hei[x] > hei[y]) {
            swap(x, y); // 选取深度大的节点做有用节点
        }
        sum += 1ll * z * (1ll * d[y] * (n - f[y]) + 1ll * (d[root] - d[y]) * f[y]);
    }
    printf("%lld", sum);
}

} // namespace STDquantum

int main() {

#ifndef ONLINE_JUDGE
    freopen("sign.in", "r", stdin);
    freopen("sign.out", "w", stdout);
#endif

    STDquantum::main();
    return 0;
}
```

# $Map$

$(map.cpp/.in/.out)$

$Time\ limit:1s$

$Memory\ limit:64M$

## $Description$

一张 $N$ 个点的无向完全图，从 $1$ 号点出发，每秒随机沿当前点的出度移动一条边。

$Q$ 次询问，每次给出一个数 $t_i$，询问第 $t_i$ 秒在 $1$ 号点的概率，答案对 $998244353$ 取模。

## $Input$

第一行是两个整数 $N,Q$。

接下来 $Q$ 行，每行一个整数，第 $i+1$ 行的数字代表 $t_i$。

## $Output$

输出 $Q$ 行，每行一个非负整数，第 $i$ 行的数字代表第 $i$ 次询问的结果。

## $Sample$

|     $Input$     |    $Output$    |
| :-------------: | :------------: |
| $3\ 2$<br>$0$<br>$2$ | $1$<br>$499122177$ |

## $Hint$

|  $ID$   |     $N$      |    $Q$    |    $t_i$     |
| :-----: | :----------: | :-------: | :----------: |
|   $1$   |     $=2$     | $\le10^5$ | $\le10^{18}$ |
|  $2-6$  |    $\le5$    | $\le10^5$ |    $\le5$    |
| $7-12$  |  $\le10^5$   | $\le10^5$ |  $\le10^5$   |
| $13-16$ | $\le10^{18}$ |   $=1$    | $\le10^{18}$ |
| $17-20$ | $\le10^{18}$ | $\le10^5$ | $\le10^{18}$ |

## 样例

### 输入#1

```
3 2
0
2
```

### 输出#1

```
1
499122177
```

## 做题

试着打个表

- $N=2$

  ```
  t = 0 -> ans = 1
  t = 1 -> ans = 0
  ```

- $N=3$

  ```
  t = 0 -> ans = 1
  t = 1 -> ans = 0
  t = 2 -> ans = 1/2
  t = 3 -> ans = 1/4
  t = 4 -> ans = 1/4
  ```

- $N=4$

  ```
  t = 0 -> ans = 1
  t = 1 -> ans = 0
  t = 2 -> ans = 1/3
  t = 3 -> ans = 2/9
  t = 4 -> ans = 20/81
  ```

勉强把$t=0,1,2,3,4$的都推出来，
$$
t=0,ans=1\\t=1,ans=0\\t=2,ans=\frac1{n-1}\\t=3,ans=\frac{n-2}{(n-1)^2}\\t=4,ans=\frac{3(n-1)+(n\ge4)?(n-2)(n-3):0}{3(n-1)^3}
$$

打完表就十分诶

正解要考虑$f[i]$表示第$i$秒在$1$的概率，$g[i]$表示不在$i$的概率。

{% asset_image 1.png %}

如图，如果当前在$1$那么从$1$到剩下的地方的话概率是$100\%$，如果在剩下的地方，有$\frac1{n-1}$的概率回到$1$，剩下$\frac{n-2}{n-1}$的概率待在剩下的这一片区域里。

所以可以得到递推方程
$$
f[i]=g[i - 1]\times\frac1{n-1}\\
g[i]=f[i-1]+g[i-1]\times\frac{n-2}{n-1}
$$
其实有了这两个方程$60$分就没问题了，当然再推一推就成正解了。

尝试把$f[i-1]$代入$g[i]$的方程中，可以得到
$$
g[i]=g[i-2]\times\frac{1}{n-1}+g[i-1]\times\frac{n-2}{n-1}
$$
对高中数学熟悉的小朋友们应该注意到了这式子的有趣性。进一步化简
$$
g[i]=g[i-1]\times\left(1-\frac1{n-1}\right)+g[i-2]\times\frac1{n-1}\\
g[i]-g[i-1]=\left(-\frac1{n-1}\right)\left(g[i-1]-g[i-2]\right)
$$
于是{$g[i]-g[i-1]$}便成了首项$g[1]-g[0]=1-0=1$，公比为$-\frac1{n-1}$的等比数列，于是便可由等比数列求和公式得到
$$
g[i]-g[0]=g[i]=\frac{1\times\left(1-\left(-\frac{1}{n-1}\right)^i\right)}{1-\left(-\frac1{n-1}\right)}=\frac{1-\left(-\frac{1}{n-1}\right)^i}{\frac n{n-1}}=\frac{(n-1)^i+(-1)^{i+1}}{n\times(n-1)^{i-1}}
$$
所以根据$f[i]$的递推式，可以得到
$$
f[i]=\frac{(n-1)^{i-1}+(-1)^i}{n\times(n-1)^{i-1}}
$$
用快速幂求解即可。

代码

```cpp
#include <bits/stdc++.h>
using namespace std;

namespace STDquantum {

#define mop(x, y) (1ll * ((x) % mod) * ((y) % mod) % mod)
#define aop(x, y) ((0ll + (x + mod) % mod + (y + mod) % mod) % mod)
typedef long long ll;
const ll N = 1e5 + 10, mod = 998244353;
ll n, q, f[N], t;
ll fpm(ll a, ll b) {
	ll ans = 1;
	a %= mod;
	for(; b; b >>= 1, a = mop(a, a)) {
		if (b & 1) ans = mop(ans, a);
	}
	return ans;
}
ll solve(ll a) {
	ll up = aop(fpm(n - 1, a - 1), (a & 1) ? -1 : 1);
	ll down = mop(n, fpm(n - 1, a - 1));
	return mop(up, fpm(down, mod - 2));
}
inline void main() {
	scanf("%lld%lld", &n, &q);
	while (q--) {
		scanf("%lld", &t);
		if (t == 1) {
			puts("0");
		} else if (t == 0) {
			puts("1");
		} else 
		printf("%lld\n", solve(t));
	}
}

} // namespace STDquantum

int main() {
	STDquantum::main();
	return 0;
}
```

## $Description$

一张 $N$ 个节点的无向图，边的情况按照以下 $M$ 条形如 $l_1\ r_1\ l_2\ r_2$ 的信息给出：

代表编号区间 $[l_1,r_1]$ 内的每一个点与编号区间 $[l_2,r_2]$ 内的每一个点之间都有一条长度为 $1$ 的边。

求 $S$ 点的单源最短路，保证图连通，给出信息中可能有很多重边。

## $Input$

第一行是三个整数 $N,M,S$。

接下来 $M$ 行，每行四个整数 $l_1[i],r_1[i],l_2[i],r_2[i]$，表示存在按题目叙述的边。

## $Output$

## $Sample$

|                    $Input$                    |    $Output$     |

| :-----: | :--------------: | :-------: |
|  $1-6$  |     $\le20$      |  $\le20$  |
| $ 7-12$ |    $\le5000$     | $\le10^5$ |
| $13-20$ | $\le5\times10^5$ | $\le10^5$ |

## $Solution$

这题我先放个标程，具体的线段树优化建图的笔记放在一个新的博客里来讲。

<a href="../xian-duan-shu-you-hua-jian-tu">线段树优化建图</a>

```cpp
#include <bits/stdc++.h>
#define LOCAL LOCAL
using namespace std;

namespace STDquantum {

#define intt register int
#define mid ((l + r) >> 1)
const int N = 4e6 + 10;
int n, m, s, cnt, tot, dis[N << 1], head[N << 1];
bool vis[N];
priority_queue<pair<int, int> > q;
struct E {
  int next, to, w;
} e[N << 1];
template <typename T> inline void read(T &x) {
  char ch;
  while (ch = getchar(), !isdigit(ch)) continue;
  x = (ch ^ 48);
  while (ch = getchar(), isdigit(ch)) x = x * 10 + (ch ^ 48);
}
inline void add(int from, int to, int w) {
  e[++tot] = (E){head[from], to, w};
  head[from] = tot;
}

struct In {
  int root, ptr;
  struct T {
    int ls, rs, p;
  } c[N << 1];

  inline int newnode() { return ++ptr; }
  void build(int &k, int l, int r) {
    k = newnode();
    if (l == r) {
      c[k].p = l;
      return;
    }
    build(c[k].ls, l, mid);
    build(c[k].rs, mid + 1, r);
    c[k].p = ++cnt;
    add(c[k].p, c[c[k].ls].p, 0);
    add(c[k].p, c[c[k].rs].p, 0);
  }
  void update(int k, int l, int r, int L, int R, int p, int w) {
    if (l > R || r < L) return;
    if (L <= l && r <= R) {
      add(p, c[k].p, w);
      return;
    }
    if (L <= mid) update(c[k].ls, l, mid, L, R, p, w);
    if (R > mid) update(c[k].rs, mid + 1, r, L, R, p, w);
  }
} treein;

struct Out {
  int root, ptr;
  struct T {
    int ls, rs, p;
  } c[N << 1];

  inline int newnode() { return ++ptr; }
  void build(int &k, int l, int r) {
    k = newnode();
    if (l == r) {
      c[k].p = l;
      return;
    }
    build(c[k].ls, l, mid);
    build(c[k].rs, mid + 1, r);
    c[k].p = ++cnt;
    add(c[c[k].ls].p, c[k].p, 0);
    add(c[c[k].rs].p, c[k].p, 0);
  }
  void update(int k, int l, int r, int L, int R, int p, int w) {
    if (l > R || r < L) return;
    if (L <= l && r <= R) {
      add(c[k].p, p, w);
      return;
    }
    if (L <= mid) update(c[k].ls, l, mid, L, R, p, w);
    if (R > mid) update(c[k].rs, mid + 1, r, L, R, p, w);
  }
} treeout;

inline void Dijkstra() {
  memset(dis, 0x3f, sizeof(dis));
  dis[s] = 0;
  q.push(make_pair(0, s));
  while (!q.empty()) {
    intt u = q.top().second;
    q.pop();
    if (vis[u]) continue;
    vis[u] = 1;
    for (intt i = head[u], v; i; i = e[i].next) {
      if (dis[v = e[i].to] > dis[u] + e[i].w) {
        dis[v] = dis[u] + e[i].w;
        q.push(make_pair(-dis[v], v));
      }
    }
  }
}
inline void main() {
  read(n), read(m), read(s);
  cnt = n;
  treein.build(treein.root, 1, n);
  treeout.build(treeout.root, 1, n);
  for (int i = 1, l1, r1, l2, r2; i <= m; ++i) {
    read(l1), read(r1), read(l2), read(r2);
    ++cnt;
    treein.update(treein.root, 1, n, l2, r2, cnt, 1);
    treeout.update(treeout.root, 1, n, l1, r1, cnt, 1);
    ++cnt;
    treein.update(treein.root, 1, n, l1, r1, cnt, 1);
    treeout.update(treeout.root, 1, n, l2, r2, cnt, 1);
  }
  Dijkstra();
  for (intt i = 1; i <= n; ++i) { printf("%d\n", dis[i] >> 1); }
}

}  // namespace STDquantum

int main() {
#ifndef ONLINE_JUDGE
#ifdef LOCAL
  freopen("test.in", "r", stdin);
  freopen("test.out", "w", stdout);
#endif
#ifndef LOCAL
  freopen("T139273_Path.in", "r", stdin);
  freopen("T139273_Path.out", "w", stdout);
#endif
#endif
  STDquantum::main();
  return 0;
}
```
