# P4158 BZOJ1296 [SCOI2009] 粉刷匠

## Description

windy有 $N$ 条木板需要被粉刷。 每条木板被分为 $M$ 个格子。 每个格子要被刷成红色或蓝色。 windy每次粉刷，只能选择一条木板上一段连续的格子，然后涂上一种颜色。 每个格子最多只能被粉刷一次。 如果windy只能粉刷 $T$ 次，他最多能正确粉刷多少格子？ 一个格子如果未被粉刷或者被粉刷错颜色，就算错误粉刷。

## Input

输入文件paint.in

第一行包含三个整数，$N,M,T$。

接下来有$N$行，每行一个长度为$M$的字符串，`0`表示红色，`1`表示蓝色。

## Output

输出文件paint.out

包含一个整数，最多能正确粉刷的格子数。

## Sample Input

```
3 6 3
111111
000000
001100
```

## Sample Output

```
16
```

## HINT

$30\%$的数据，满足 $1 \le N,M \le 10$ ； $0 \le T \le 100$ 。


$100\%$的数据，满足 $1 \le N,M \le 50$ ； $0\le T \le 2500$。

首先预处理出每一行的前缀和$sum[i][j]$，这样可以方便地求出一个区间里有多少个$0$或者$1$。
$$
f[t][i][j][0]=\max_{0<k<j}\left\{f[t-1][i][k][0/1]+(j-k-(sum[i][j]-sum[i][k]))\right\}\\f[t][i][j][1]=\max_{0<k<j}\left\{f[t-1][i][k][0/1]+sum[i][j]-sum[i][k]\right\}
$$
不过这样好像只能在一行里面转移，而且空间似乎有点挤，可以用滚动数组压掉时间维。

多行之间的转移，把$j$作为考虑到第$j$个，那么在转移的时候要像上面一样考虑到$j$这里刷了，也要想到到$j$这里就没刷，直接继承上一个的$f[t][i][k][0/1]$，不过这个$0/1$好像有点鸡肋？？

f[t][i][j]=\max\left\{f[t][i][j-1],\max_{0<k<j}\left\{f[t-1][i][k]+\max\left\{j-k-S[j][k],S[j][k]\right\} \right\} \right\}

```cpp
#include <algorithm>
#include <cctype>
#include <cstdio>
using namespace std;

namespace STDquantum {

template <typename T> inline void read(T &x) {
  char ch;
  while (ch = getchar(), !isdigit(ch)) continue;
  x = (ch ^ 48);
  while (ch = getchar(), isdigit(ch)) x = x * 10 + (ch ^ 48);
}
const int N = 51;
int n, m, t, sum[N][N], f[2][N][N];
char str[N];
inline void main() {
  read(n), read(m), read(t);
  if (t >= n * m) { return (void)printf("%d\n", n * m); }
  for (register int i = 1; i <= n; ++i) {
    scanf("%s", str + 1);
    for (register int j = 1; j <= m; ++j) {
      sum[i][j] = sum[i][j - 1] + (str[j] == '1');
    }
  }
  for (register int T = 1; T <= t; ++T) {
    register int now = T & 1, las = (T & 1) ^ 1;
    for (register int i = 1; i <= n; ++i) {
      f[las][i][0] = f[now][i][0] = f[las][i - 1][m];
      for (register int j = 1; j <= m; ++j) {
        f[now][i][j] = f[now][i][j - 1];
        for (register int k = 0; k < j; ++k) {
          f[now][i][j] = max(
              f[now][i][j],
              f[las][i][k] + max(sum[i][j] - sum[i][k],
                                 j - k - sum[i][j] + sum[i][k]));
        }
      }
    }
  }
  printf("%d\n", f[t & 1][n][m]);
}

}  // namespace STDquantum

int main() {
  STDquantum::main();
  return 0;
}
```

真就每个晚上过夜？到底一天要安慰几个？一点没谈。

样例解释也没有就尼玛离谱。

其实就是个克鲁斯卡尔求最小生成树，每个边经过两次，每条边把起点的点权加上。求完最小生成树加上最小的点权就行。。什么乱七八糟的。

网上看的`Find`一行函数锅了，果然这玩意还是自己写的好使。

```cpp
#include <bits/stdc++.h>
using namespace std;

namespace STDquantum {

const int N = 1e4 + 10, M = 1e5 + 10;
int n, m, minn, w[N], fa[N];
struct E {
  int from, to, w;
  inline bool operator<(const E &rhs) const {
    return w < rhs.w;
  }
} e[M];
inline void UFset() {
  for (int i = 1; i <= n; ++i) { fa[i] = -1; }
}
int Find(int x) {
  int s;
  for (s = x; fa[s] >= 0; s = fa[s]) continue;
  while (s != x) {
    int tmp = fa[x];
    fa[x] = s;
    x = tmp;
  }
  return s;
}
inline void Union(int x, int y) {
  int R1 = Find(x), R2 = Find(y);
  if (fa[R1] > fa[R2]) {
    fa[R1] += fa[R2];
    fa[R2] = R1;
  } else {
    fa[R2] += fa[R1];
    fa[R1] = R2;
  }
}
void Kruskal() {
  int wei = 0, sum = 0;
  UFset();
  for (int i = 1, u, v; i <= m; ++i) {
    u = e[i].from, v = e[i].to;
    if (Find(u) != Find(v)) {
      wei += e[i].w, ++sum;
      Union(u, v);
    }
    if (sum >= n - 1) break;
  }
  printf("%d", wei + minn);
}
inline void main() {
  scanf("%d%d", &n, &m);
  minn = 0x7fffffff;
  for (int i = 1; i <= n; ++i) {
    scanf("%d", &w[i]);
    minn = min(minn, w[i]);
  }
  for (int i = 1, x, y, z; i <= m; ++i) {
    scanf("%d%d%d", &x, &y, &z);
    e[i] = (E){x, y, z * 2 + w[x] + w[y]};
  }
  sort(e + 1, e + m + 1);
  Kruskal();
}

}  // namespace STDquantum

int main() {
  STDquantum::main();
  return 0;
}
```

# P2896 [USACO08FEB]Eating Together S

## 做题

设$f[i][1/2/3]$代表已经处理完了序列$[1,i-1]$的部分，现在到了第$i$个，第$i$个是$1/2/3$，总共改变最小的次数。

转移简单的一批，一遍写对。

```cpp
#include <bits/stdc++.h>
using namespace std;

namespace STDquantum {

const int N = 3e5 + 10;
int n, ans, a[N], f[N][4];
inline void main() {
  scanf("%d", &n);
  for (int i = 1; i <= n; ++i) {
    scanf("%d", &a[i]);
    if (a[i] == 1) {
      f[i][1] = f[i - 1][1];
      f[i][2] = min(f[i - 1][2], f[i - 1][1]) + 1;
      f[i][3] = 1 + min(f[i - 1][1],
                        min(f[i - 1][2], f[i - 1][3]));
    } else if (a[i] == 2) {
      f[i][1] = f[i - 1][1] + 1;
      f[i][2] = min(f[i - 1][1], f[i - 1][2]);
      f[i][3] = 1 + min(f[i - 1][3],
                        min(f[i - 1][1], f[i - 1][2]));
    } else {
      f[i][1] = f[i - 1][1] + 1;
      f[i][2] = 1 + min(f[i - 1][1], f[i - 1][2]);
      f[i][3] =
          min(f[i - 1][3], min(f[i - 1][2], f[i - 1][1]));
    }
  }
  ans = min(f[n][1], min(f[n][2], f[n][3]));
  memset(f, 0, sizeof(f));
  for (int i = 1; i <= n; ++i) {
    if (a[i] == 1) {
      f[i][1] =
          min(f[i - 1][3], min(f[i - 1][2], f[i - 1][1]));
      f[i][2] = min(f[i - 1][2], f[i - 1][3]) + 1;
      f[i][3] = f[i - 1][3] + 1;
    } else if (a[i] == 2) {
      f[i][1] = 1 + min(f[i - 1][3],
                        min(f[i - 1][1], f[i - 1][2]));
      f[i][2] = min(f[i - 1][3], f[i - 1][2]);
      f[i][3] = f[i - 1][3] + 1;
    } else {
      f[i][1] = 1 + min(f[i - 1][3],
                        min(f[i - 1][2], f[i - 1][1]));
      f[i][2] = 1 + min(f[i - 1][3], f[i - 1][2]);
      f[i][3] = f[i - 1][3];
    }
  }
  ans = min(ans, min(min(f[n][1], f[n][2]), f[n][3]));
  printf("%d", ans);
}

}  // namespace STDquantum
```

【输入输出样例 1 说明】

## 做题

我们设$f[i][j]$表示第$i$个人等了$j$分钟，且全都上过车的情况下，前$i$个人总的等待时间。

那么考虑转移，枚举一个$k$代表第$k$个人，考虑如何求出$f[k][time]$。

首先第$i$个人开始等待的时间是$t[i]$，然后等了$j$，上了车以后，车又走了$m$，所以$time=\max(t[i]+j+m-t[k],0)$，和$0$取最大是因为如果前面那个式子的值为负，也就是说$t[k]>t[i]+j+m$，就是$k$来的时刻比车回来的时刻还晚， 那么就不用等了。

然后因为第$k$个人在$t[k]+time$的时候上车，之前$k-i$个人总的等待时间是$$\sum_{p=i+1}^kt[k]+time-t[p]$$，前面两部分是常数，提出来，得到$$(t[k]+time)\times(k-i)-\sum_{p=i+1}^kt[p]$$。再用前缀和整一下，总的DP方程就是
$$
f[k][time]=f[i][j]+(t[k]+time)\times(k-i)-sum[k]+sum[i]
$$

```cpp
#include <bits/stdc++.h>
using namespace std;

namespace STDquantum {

const int N = 510, inf = 0x3f3f3f3f;
int n, m, ans, t[N], sum[N], f[N][N];
inline void main() {
  scanf("%d%d", &n, &m);
  for (int i = 1; i <= n; ++i) { scanf("%d", &t[i]); }
  sort(t + 1, t + n + 1);
  for (int i = 1; i <= n; ++i) {
    sum[i] = t[i] + sum[i - 1];
  }
  memset(f, 0x3f, sizeof(f));
  t[0] = -0x3f3f3f3f;
  f[0][0] = 0;
  for (int i = 0; i <= n; ++i) {
    int minn = min(t[i + 1] - t[i], m - 1);
    for (int j = 0; j <= minn; ++j) {
      if (f[i][j] != inf) {
        for (int k = i + 1; k <= n; ++k) {
          int time = max(t[i] + j + m - t[k], 0);
          f[k][time] =
              min(f[i][j] + (t[k] + time) * (k - i) -
                      sum[k] + sum[i],
                  f[k][time]);
        }
      }
    }
  }
  ans = inf;
  for (int i = 0; i < m; ++i) { ans = min(ans, f[n][i]); }
  printf("%d", ans);
}

}  // namespace STDquantum

int main() {
  STDquantum::main();
  return 0;
}
```
