```cpp
#include <bits/stdc++.h>

using namespace std;

#define ge getchar()
#define Re read()

typedef long long ll;

inline int read() {
  int t = 0, x = 0, ch;
  while (!isdigit(ch = ge)) t |= ch == '-';
  while (isdigit(ch)) x = (x << 1) + (x << 3) + (ch ^ 48), ch = ge;
  return t ? -x : x;
}

template <typename T> inline void chkmin(T& a, T b) {
  a = a < b ? a : b;
}

const int MAXN = 100010;
const ll Inf = 1LL << 60;

struct Pill {
  ll a, b, c;

  friend bool operator<(Pill a, Pill b) { return a.c > b.c; }
} pill[MAXN];
ll c[MAXN];
int res = 1 << 30, N;
ll L;

struct BIT {
  ll T[MAXN];

  inline void Add(int p, ll v) {
    while (p <= N) T[p] += v, p += p & -p;
  }
  inline ll Query(int p) {
    ll v = 0;
    while (p) v += T[p], p ^= p & -p;
    return v;
  }
  inline int find(ll v) {
    int l = 1, r = N, res = N + 1;
    while (l <= r) {
      int mid = (l + r) >> 1;
      if (Query(mid) >= v)
        res = mid, r = mid - 1;
      else
        l = mid + 1;
    }
    return res;
  }
} bit;

struct Segment_Tree {
#define ls (x << 1)
#define rs (x << 1 | 1)

  struct Node {
    ll mi, tg;
  } T[MAXN << 2];

  inline void pushdown(int x) {
    ll tg = T[x].tg;
    if (!tg) return;
    T[ls].mi += tg, T[ls].tg += tg;
    T[rs].mi += tg, T[rs].tg += tg;
    T[x].tg = 0;
  }

  inline void build(int x, int l, int r) {
    if (l == r) return T[x].mi = -c[l], void();
    int mid = (l + r) >> 1;
    build(ls, l, mid);
    build(rs, mid + 1, r);
    T[x].mi = min(T[ls].mi, T[rs].mi);
  }

  inline void Modify(int x, int l, int r, int L, int R, int v) {
    if (L <= l && r <= R) return T[x].mi += v, T[x].tg += v, void();
    pushdown(x);
    int mid = (l + r) >> 1;
    if (L <= mid) Modify(ls, l, mid, L, R, v);
    if (mid < R) Modify(rs, mid + 1, r, L, R, v);
    T[x].mi = min(T[ls].mi, T[rs].mi);
  }

  inline ll Query(int x, int l, int r, int L, int R) {
    if (L <= l && r <= R) return T[x].mi;
    pushdown(x);
    ll mi = Inf;
    int mid = (l + r) >> 1;
    if (L <= mid) chkmin(mi, Query(ls, l, mid, L, R));
    if (mid < R) chkmin(mi, Query(rs, mid + 1, r, L, R));
    return mi;
  }

#undef ls
#undef rs
} seg;

inline void solve(int l, int r, int L, int R) {
  if (l > r) return;
  if (l == r) {
    int pos = bit.find(::L - pill[l].a);
    if (seg.Query(1, 1, N, 1, pos) > 0) chkmin(res, pos + 1);
    return;
  }
  int mid = (l + r) >> 1, tL, tR;
  tL = L;
  for (int i = l; i <= mid; i++) {
    seg.Modify(1, 1, N, tL, N, pill[i].c);
    bit.Add(tL, pill[i].c), tL++;
  }
  solve(mid + 1, r, tL, R);
  tL = L;
  for (int i = l; i <= mid; i++) {
    seg.Modify(1, 1, N, tL, N, -pill[i].c);
    bit.Add(tL, -pill[i].c), tL++;
  }
  tR = R;
  for (int i = r; i > mid; i--) {
    seg.Modify(1, 1, N, tR, N, pill[i].c);
    bit.Add(tR, pill[i].c), tR--;
  }
  solve(l, mid, L, tR);
  tR = R;
  for (int i = r; i > mid; i--) {
    seg.Modify(1, 1, N, tR, N, -pill[i].c);
    bit.Add(tR, -pill[i].c), tR--;
  }
}

int main() {
  int n = Re;
  L = Re;
  for (int i = 1; i <= n; i++)
    pill[i].c = (pill[i].a = Re) - (pill[i].b = Re);
  for (int i = 1; i <= n; i++) c[i] = Re;
  sort(pill + 1, pill + 1 + n), N = n - 1;
  for (int i = 1; i <= N; i++) c[i] += c[i - 1];
  seg.build(1, 1, N), solve(1, n, 1, N);
  printf("%d\n", res > n ? -1 : res);
  return 0;
}
```

```cpp
#include <bits/stdc++.h>

using namespace std;

#define ge getchar()
#define Re read()

inline int read() {
  int t = 0, x = 0, ch;
  while (!isdigit(ch = ge)) t |= ch == '-';
  while (isdigit(ch)) x = (x << 1) + (x << 3) + (ch ^ 48), ch = ge;
  return t ? -x : x;
}

const int MAXN = 410;

int fa[MAXN];
int sz[MAXN];

inline int find(int x) {
  return fa[x] == x ? x : fa[x] = find(fa[x]);
}
inline void Merge(int x, int y) {
  int fx = find(x), fy = find(y);
  if (fx == fy) return;
  fa[fx] = fy, sz[fy] += sz[fx];
}

int main() {
#ifndef ONLINE_JUDGE
  freopen("coin.in", "r", stdin);
  freopen("coin.out", "w", stdout);
#endif
  int Case = Re;
  while (Case--) {
    int n = Re, m = Re, N = n + m;
    for (int i = 1; i <= N; i++) fa[i] = i, sz[i] = 1;
    for (int i = 1; i <= N; i++) fa[i + N] = i + N, sz[i + N] = 0;
    for (int i = 1; i <= n; i++) {
      char str[110];
      scanf("%s", str + 1);
      for (int j = 1; j <= m; j++) {
        if (str[j] == 'e') continue;
        int a1 = i, a0 = i + N;
        int b1 = j + n, b0 = j + n + N;
        if (str[j] == 'o')
          Merge(a1, b1), Merge(a0, b0);
        else
          Merge(a1, b0), Merge(a0, b1);
      }
    }
    int flg = 0, res = 0;
    for (int i = 1; i <= N; i++) {
      if (find(i) == find(i + N)) {
        flg = -N;
        break;
      }
      if (find(i) == i) {
        flg += (sz[find(i)] & 1) ^ (sz[find(i + N)] & 1);
        if (!((sz[find(i)] & 1) ^ (sz[find(i + N)] & 1)))
          res ^= sz[find(i)] & 1;
      }
    }
    if (flg < 0)
      printf("%d\n", -flg & 1);
    else
      printf("%d\n", ((res | flg) & 1) | 2);
  }
  return 0;
}
```

## merge

```cpp
#include <bits/stdc++.h>

using namespace std;

#define ge getchar()
#define Re read()

inline int read() {
  int t = 0, x = 0, ch;
  while (!isdigit(ch = ge)) t |= ch == '-';
  while (isdigit(ch)) x = (x << 1) + (x << 3) + (ch ^ 48), ch = ge;
  return t ? -x : x;
}

template <typename T> inline void chkmax(T &a, T b) {
  a = a > b ? a : b;
}

const int MAXN = 610;

int a[MAXN];
int pos[MAXN];
int las[MAXN];
int cnt[MAXN][MAXN];
int f[MAXN][MAXN];
int g[MAXN][MAXN];

inline int solve(int l, int r) {
  if (l == r) return 0;
  if (g[l][r]) return f[l][r];
  int &F = f[l][r];
  g[l][r] = 1;
  for (int i = l; i < r; i++)
    chkmax(F,
           solve(l, i) + solve(i + 1, r) + cnt[l][i] * cnt[i + 1][r]);
  return F;
}

int main() {
#ifndef ONLINE_JUDGE
  freopen("merge.in", "r", stdin);
  freopen("merge.out", "w", stdout);
#endif
  int n = Re, N = n << 1, res = 0;
  for (int i = 1; i <= n; i++) a[i] = a[i + n] = Re;
  for (int i = 1; i <= N; i++) las[i] = pos[a[i]], pos[a[i]] = i;
  for (int i = 1; i <= N; i++)
    for (int j = i; j <= N; j++)
      cnt[i][j] = cnt[i][j - 1] + (las[j] < i);
  for (int i = 1; i <= n; i++) chkmax(res, solve(i, i + n - 1));
  printf("%d\n", res);
  return 0;
}
```
