# 排列

## 题目描述

小 A 是一个爱好排列的女孩子，如果一个排列 $P$ 满足对于所有的 $i$ 都有：$\vert P_i -i\vert \neq  k$。那么我们就称排列 $P$ 是可爱的。 对于小 A 给定的 $n$ 和 $k$，请问有多少种不同的可爱的排列？ 因为这个数量实在太大了，请输出答案对 $998244353$ 取模后的结果。

## 输入输出格式

### 输入格式

第一行两个正整数 $n，k$，含义如题面所示。

### 输出格式

一行一个整数，表示可爱排列的数量对 $998244353$ 取模后的结果。

## 输入输出样例

### 输入样例 #1

```
3 1
```

### 输出样例 #1

```
2
```

## 说明

对于 $20\%$ 的数据，$1 \leq n \leq10$ 。

对于 $40\%$ 的数据，$1 \leq n \leq 20$ 。

对于 $60\%$ 的数据，$1 \leq n \leq 100$ 。

对于 $95\%$ 的数据，$1 \leq n \leq 2,000$ 。

对于 $100\%$ 的数据，$1 \leq n \leq100,000 ，1\leq k \leq n -1$ 。

## Std

```cpp
#include <bits/stdc++.h>
#define LL long long
#define F(x, y, z) for (int x = y; x <= z; ++x)
#define D(x, y, z) for (int x = y; x >= z; --x)
using namespace std;
const int N = 1000010;
LL R() {
  LL ans = 0, f = 1;
  char c = ' ';
  for (; c < '0' || c > '9'; c = getchar())
    if (c == '-') f = -1;
  for (; c >= '0' && c <= '9'; c = getchar())
    ans = ans * 10 + c - '0';
  return ans * f;
}
char s[N];
int n;
int p[26][N], fn[N];
LL ans;
int Find(int k, int x) {
  return p[k][x] == x ? x : (p[k][x] = Find(k, p[k][x]));
}
int main() {
  freopen("C.in", "r", stdin);
  freopen("C.out", "w", stdout);
  cin >> (s + 1);
  n = strlen(s + 1);
  F(i, 1, n) s[i] -= 'a';
  F(k, 0, 25) F(i, 0, n + 1) p[k][i] = i;
  s[n + 1] = s[n + 2] = 26;
  D(i, n, 1) {
    if (s[i] == s[i + 1]) {
      F(k, 0, 25)
      p[k][i] = i + 2;
      p[s[i]][i] = i;
      fn[i] = fn[i + 2] + 1;
    } else {
      int nx = Find(s[i], i + 1);
      if (s[i] == s[nx]) {
        F(k, 0, 25)
        p[k][i] = nx + 1;
        p[s[i]][i] = i;
        fn[i] = fn[nx + 1] + 1;
      }
    }
    ans += fn[i];
  }
  ans %= 998244353;
  cout << ans << '\n';
  return 0;
}
```

# 苹果树

## 题目描述

小 B 是一个辛勤的农民，他家里种了一棵很大的苹果树。

这棵苹果树可以看作一张 n 个点 n-1 条边的无向连通图，小 B 觉得这颗苹果树很脆弱， 因为只要剪断任意一条边，苹果树就不连通了，于是他给苹果树新加了 m 条边。

现在这颗苹果树就不像是一棵树了，成了一张 n 个点 n+m-1 条边的无向连通图，小 Q 是小 B 的好朋友，他觉得这棵树依然很脆弱，他告诉小 B，自己只要破坏两条边，一条是原树中的边，一条是小 B 新加的边，就能使苹果树不连通。

小 B 觉得小 Q 说得很不可思议，于是他找来了擅长编程的你，希望你告诉他，按照小 B 的破坏规则，有多少种不同的方案使得苹果树不连通。

注意：两种方案不同当且仅当一条边在第一种方案中被删除了但在第二种方案中没有被删除。

## 输入输出格式

### 输入格式

第一行两个正整数 n，m，表示苹果树的点数和后面新加的边的个数。接下来 n-1 行，每行两个正整数 u,v，表示(u,v)之间有一条树边。 接下来 m 行，每行两个正整数 u,v，表示(u,v)之间有一条小 B 新加的边。

### 输出格式

输出一行一个整数，表示有多少种不同的方案。

## 输入输出样例

### 输入样例 #1

```
4 1
1 2
2 3
1 4
3 4
```

### 输出样例 #1

```
3
```

## 说明

对于 30% 的数据，$1 \leq n \leq 5 ，1 \leq m \leq5$ 。

对于 60% 的数据，$1 \leq n \leq 1,000 ，1 \leq m \leq 1,000$ 。 

对于 90% 的数据，$1 \leq n \leq 100,000 ，1 \leq m \leq 100,000$ 。

对于 100% 的数据，$1 \leq n \leq 300,000 ，1 \leq m \leq 300,000$ 。

## solution

考试时候考虑的是计算每条边被非树边覆盖的次数，如果为 $0$ 答案加 $m$，如果为 $1$ 答案加 $1$。然后就树剖套线段树。完美TLE了。

开O2能过（正解差分不香吗？）

```cpp
#include <bits/stdc++.h>
using namespace std;

namespace STDquantum {

template <typename T> inline void read(T &x) {
  char ch;
  int f = 1;
  while (ch = getchar(), !isdigit(ch))
    if (ch == '-') f = -1;
  x = (ch ^ 48);
  while (ch = getchar(), isdigit(ch)) x = x * 10 + (ch ^ 48);
  x *= f;
}

int n, m;

const int N = 2e6 + 10;
int tot, head[N];
struct E {
  int next, to;
} e[N];
inline void add(int x, int y) {
  e[++tot] = (E){head[x], y}, head[x] = tot;
  e[++tot] = (E){head[y], x}, head[y] = tot;
}

int fa[N], dep[N], sz[N], son[N], dfn[N], top[N];
void dfs1(int s, int f) {
  fa[s] = f, dep[s] = dep[f] + 1, sz[s] = 1;
  for (int i = head[s], v; i; i = e[i].next) {
    if ((v = e[i].to) == f) continue;
    dfs1(v, s), sz[s] += sz[v];
    if (sz[v] > sz[son[s]]) son[s] = v;
  }
}
void dfs2(int s, int t) {
  top[s] = t, dfn[s] = ++tot;
  if (son[s]) dfs2(son[s], t);
  for (int i = head[s], v; i; i = e[i].next) {
    if ((v = e[i].to) != son[s] && v != fa[s]) dfs2(v, v);
  }
}

#define lson (k << 1)
#define rson (k << 1 | 1)
#define mid ((l + r) >> 1)
struct T {
  int l, r, lz, val;
} t[N];
inline void pushup(int k) { t[k].val = t[lson].val + t[rson].val; }
inline int len(int k) { return t[k].r - t[k].l + 1; }
inline void pushdown(int k) {
  if (!t[k].lz) return;
  t[lson].lz += t[k].lz, t[rson].lz += t[k].lz;
  t[lson].val += len(lson) * t[k].lz;
  t[rson].val += len(rson) * t[k].lz;
  t[k].lz = 0;
}
void build(int k, int l = 1, int r = n) {
  t[k].l = l, t[k].r = r;
  if (l == r) return;
  build(lson, l, mid), build(rson, mid + 1, r);
  pushup(k);
}
void modify(int k, int L, int R, int w, int l = 1, int r = n) {
  if (L <= l && r <= R) {
    return t[k].val += w * len(k), t[k].lz += w, void();
  }
  pushdown(k);
  if (mid >= L) modify(lson, L, R, w, l, mid);
  if (mid < R) modify(rson, L, R, w, mid + 1, r);
  pushup(k);
}
int query(int k, int pos, int l = 1, int r = n) {
  if (l == r) return t[k].val;
  pushdown(k);
  if (mid >= pos) {
    return query(lson, pos, l, mid);
  } else {
    return query(rson, pos, mid + 1, r);
  }
}
void modify1(int x, int y, int w) {
  while (top[x] != top[y]) {
    if (dep[top[x]] < dep[top[y]]) swap(x, y);
    modify(1, dfn[top[x]], dfn[x], w);
    x = fa[top[x]];
  }
  if (dfn[x] > dfn[y]) swap(x, y);
  modify(1, dfn[x], dfn[y], w);
}
inline int lca(int x, int y) {
  while (top[x] != top[y]) {
    if (dep[top[x]] < dep[top[y]]) swap(x, y);
    x = fa[top[x]];
  }
  return dep[x] < dep[y] ? x : y;
}

long long ans;
void dfs(int s) {
  for (int i = head[s], v, tmp; i; i = e[i].next) {
    if ((v = e[i].to) == fa[s]) continue;
    dfs(v), tmp = query(1, dfn[v]);
    if (!tmp) ans += m;
    if (tmp == 1) ++ans;
  }
}

inline void main() {
  read(n), read(m);
  for (int i = 1, x, y; i < n; ++i) read(x), read(y), add(x, y);
  tot = 0, dfs1(1, 0), dfs2(1, 1), build(1);
  for (int i = 1, x, y, z; i <= m; ++i) {
    read(x), read(y), z = lca(x, y);
    modify1(x, y, 1), modify1(z, z, -1);
  }
  dfs(1);
  printf("%lld", ans);
}

}  // namespace STDquantum

int main() {
#ifndef ONLINE_JUDGE
#ifdef LOCAL
  freopen("D:\\OI\\STDquantum\\test.in", "r", stdin);
  freopen("D:\\OI\\STDquantum\\test.out", "w", stdout);
#endif
#ifndef LOCAL
  freopen("tree.in", "r", stdin);
  freopen("tree.out", "w", stdout);
#endif
#endif
  STDquantum::main();
  return 0;
}

#include <bits/stdc++.h>

using namespace std;

#define mp make_pair
#define pb push_back
#define fi first
#define se second
#define ALL(x) (x).begin(), (x).end()
#define SZ(x) ((int)((x).size()))

typedef pair<int, int> PII;
typedef vector<int> VI;
typedef long long int64;
typedef unsigned int uint;
typedef unsigned long long uint64;

#define gi(x) ((x) = F())
#define gii(x, y) (gi(x), gi(y))
#define giii(x, y, z) (gii(x, y), gi(z))

int F() {
  char ch;
  int x, a;
  while (ch = getchar(), (ch < '0' || ch > '9') && ch != '-')
    ;
  if (ch == '-')
    ch = getchar(), a = -1;
  else
    a = 1;
  x = ch - '0';
  while (ch = getchar(), ch >= '0' && ch <= '9')
    x = (x << 1) + (x << 3) + ch - '0';
  return a * x;
}

const int N = 3e5 + 10;

int n, m;

struct edge {
  int to;
  edge *next;
} e[N << 1], *et = e, *la[N];

void add(int u, int v) {
  *++et = (edge){v, la[u]}, la[u] = et;
  *++et = (edge){u, la[v]}, la[v] = et;
}

int siz[N], son[N], fa[N], dep[N];

void dfs1(int u) {
  siz[u] = 1;
  dep[u] = dep[fa[u]] + 1;
  for (edge *it = la[u]; it; it = it->next) {
    int v = it->to;
    if (v == fa[u]) continue;
    fa[v] = u;
    dfs1(v);
    siz[u] += siz[v];
    if (siz[son[u]] < siz[v]) son[u] = v;
  }
}

int top[N];

void dfs2(int u, int tp) {
  top[u] = tp;
  if (son[u]) dfs2(son[u], tp);
  for (edge *it = la[u]; it; it = it->next) {
    if (it->to == son[u] || it->to == fa[u]) continue;
    dfs2(it->to, it->to);
  }
}

int lca(int u, int v) {
  for (; top[u] != top[v];
       dep[top[u]] > dep[top[v]] ? u = fa[top[u]] : v = fa[top[v]])
    ;
  return dep[u] < dep[v] ? u : v;
}

int cover[N];

void dfs3(int u) {
  for (edge *it = la[u]; it; it = it->next) {
    if (it->to == fa[u]) continue;
    dfs3(it->to);
    cover[u] += cover[it->to];
  }
}

int main() {
  freopen("tree.in", "r", stdin);
  freopen("tree.out", "w", stdout);
  gii(n, m);
  for (int i = 1; i < n; ++i) {
    int u, v;
    gii(u, v);
    add(u, v);
  }
  dfs1(1);
  dfs2(1, 1);
  for (int i = 1; i <= m; ++i) {
    int s, t;
    gii(s, t);
    ++cover[s], ++cover[t];
    cover[lca(s, t)] -= 2;
  }
  dfs3(1);
  long long ans = 0;
  for (int i = 2; i <= n; ++i)
    if (cover[i] == 1)
      ++ans;
    else if (!cover[i])
      ans += m;
  printf("%lld\n", ans);
  return 0;
}
```

# 括号序列

## 题目描述

可怜不喜欢括号序列，但是她发现总是有人喜欢出括号序列的题。

为了让全世界都能感受到她的痛苦，她想要写一个转换器：它能把普通的小写字符串转换成长度相同的合法的括号序列。 在可怜的构思中，这样的转换器需要满足如下两个条件： 

1. 结果的括号序列必须要是合法的，即左右括号必须要是相匹配的。 

2. 对于一堆相匹配的左右括号，他们所在的位置原来的小写字母必须相同。

举例来说，对于字符串aabaab，()(())就是一个合法的答案，而()()()不满足第二个条件， (((())不满足第一个条件。

可怜发现对于一个小写字符串，有时候有很多满足条件的括号序列，有些时候一个都没有。

于是可怜给出了一个小写字符串，她想让你帮她算一下，有多少个不同的子串满足能转换成合法的括号序列。

## 输入输出格式

### 输入格式

输入一行一个小写字符串s。

### 输出格式

输出一行一个整数，表示满足条件的括号序列的数量，对998244353取模。

## 输入输出样例

### 输入样例 #1

```
aabaab
```

### 输出样例 #1

```
4
```

### 输入样例 #2

```
abcabcabc
```

### 输出样例 #2

```
0
```

### 输入样例 #3

```
aabbcc
```

### 输出样例 #3

```
6
```

## 说明

对于30%的数据，$\vert s\vert\le20$。

对于60%的数据，$\vert s\vert\le1000$。

对于100%的数据，$\vert s\vert\le10^6$。

## std

```cpp
#include <bits/stdc++.h>

using namespace std;

#define mp make_pair
#define pb push_back
#define fi first
#define se second
#define ALL(x) (x).begin(), (x).end()
#define SZ(x) ((int)((x).size()))

typedef pair<int, int> PII;
typedef vector<int> VI;
typedef long long int64;
typedef unsigned int uint;
typedef unsigned long long uint64;

#define gi(x) ((x) = F())
#define gii(x, y) (gi(x), gi(y))
#define giii(x, y, z) (gii(x, y), gi(z))

int F() {
  char ch;
  int x, a;
  while (ch = getchar(), (ch < '0' || ch > '9') && ch != '-')
    ;
  if (ch == '-')
    ch = getchar(), a = -1;
  else
    a = 1;
  x = ch - '0';
  while (ch = getchar(), ch >= '0' && ch <= '9')
    x = (x << 1) + (x << 3) + ch - '0';
  return a * x;
}

const int mod = 998244353;

int fpow(int a, int x) {
  int ret = 1;
  for (; x; x >>= 1) {
    if (x & 1) ret = 1LL * ret * a % mod;
    a = 1LL * a * a % mod;
  }
  return ret;
}

const int N = 1 << 20;

int g[N], t[N], ans;

int fac[N], rfac[N];

int C(int n, int m) {
  if (n < m || m < 0) return 0;
  return 1LL * fac[n] * rfac[m] % mod * rfac[n - m] % mod;
}

int f(int n, int m) { return C(n - m, m); }

int inc(int a, int b) {
  a += b;
  return a >= mod ? a - mod : a;
}

int dec(int a, int b) {
  a -= b;
  return a < 0 ? a + mod : a;
}

void dft(int *a, int n, int type) {
  for (int i = 0, j = 0; i < n; ++i) {
    if (i > j) swap(a[i], a[j]);
    for (int l = n >> 1; (j ^= l) < l; l >>= 1)
      ;
  }
  for (int i = 1; i < n; i <<= 1) {
    int m = i << 1;
    int w = fpow(3, (mod - 1) / m);
    if (type == -1) w = fpow(w, mod - 2);
    for (int j = 0; j < n; j += m) {
      int v = 1;
      for (int k = 0; k < i; ++k) {
        int x = a[j + k], y = 1LL * a[j + k + i] * v % mod;
        a[j + k] = inc(x, y);
        a[j + k + i] = dec(x, y);
        v = 1LL * v * w % mod;
      }
    }
  }
  if (type == -1) {
    int invn = fpow(n, mod - 2);
    for (int i = 0; i < n; ++i) a[i] = 1LL * a[i] * invn % mod;
  }
}

int a[N], b[N];

int main() {
  freopen("permutation.in", "r", stdin);
  freopen("permutation.out", "w", stdout);
  int n, k;
  gii(n, k);
  fac[0] = 1;
  for (int i = 1; i <= n; ++i) fac[i] = 1LL * fac[i - 1] * i % mod;
  rfac[n] = fpow(fac[n], mod - 2);
  for (int i = n; i; --i) rfac[i - 1] = 1LL * rfac[i] * i % mod;
  g[0] = 1;
  for (int i = 0; i <= (n - 1) / k; ++i) a[i] = f((n - 1) / k + 1, i);
  for (int i = 0; i <= (n - 1) / k - 1; ++i) b[i] = f((n - 1) / k, i);
  int cnt1 = 0, cnt2 = 0;
  for (int i = 1; i <= k; ++i) {
    int len = (n - i) / k;
    if (len == (n - 1) / k)
      ++cnt1;
    else
      ++cnt2;
  }
  int m = 1;
  for (; m < (n << 1); m <<= 1)
    ;
  dft(a, m, 1), dft(b, m, 1);
  for (int i = 0; i < m; ++i)
    g[i] = 1LL * fpow(a[i], cnt1 << 1) * fpow(b[i], cnt2 << 1) % mod;
  dft(g, m, -1);
  for (int i = n, fac = 1, j = (n & 1) ^ 1; ~i;
       --i, fac = 1LL * fac * (n - i) % mod, j ^= 1) {
    if (j)
      ans = (ans + 1LL * g[i] * fac) % mod;
    else
      ans = (ans - 1LL * g[i] * fac) % mod;
  }
  if (ans < 0) ans += mod;
  printf("%d\n", ans);
  return 0;
}
```
