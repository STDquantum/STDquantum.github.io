#include <bits/stdc++.h>
using namespace std;

namespace STDquantum {

template <typename T> inline void read(T &x) {
  char ch;
  int f = 1;
  while (ch = getchar(), !isdigit(ch))
    if (ch == '-') f = -1;
  x = (ch ^ 48);
  while (ch = getchar(), isdigit(ch)) x = x * 10 + (ch ^ 48);
  x *= f;
}

const int N = 2e6 + 10;

#define mid ((l + r) >> 1)
int r, p, s, cnt[3];
char a[] = {'R', 'P', 'S'};
char ch[N];
inline void checkIfImpossible() {
  int sum = r + p + s, maxx = max(max(r, p), s);
  if (maxx > sum / 2) {
    puts("IMPOSSIBLE");
    exit(0);
  }
}
void dfs(char x, int l, int r) {
  if (l == r) {
    ch[l] = x;
    if (x == 'R') {
      ++cnt[0];
    } else if (x == 'P') {
      ++cnt[1];
    } else {
      ++cnt[2];
    }
    return;
  }
  if (x == 'R') {
    dfs('R', l, mid), dfs('S', mid + 1, r);
  } else if (x == 'S') {
    dfs('P', l, mid), dfs('S', mid + 1, r);
  } else {
    dfs('P', l, mid), dfs('R', mid + 1, r);
  }
  for (int i = l; i <= mid; ++i) {
    if (ch[i] < ch[mid + i - l + 1]) return;
    if (ch[i] > ch[mid + i - l + 1]) break;
  }
  for (int i = l; i <= mid; ++i) swap(ch[i], ch[mid + i - l + 1]);
}
inline void main() {
  read(r), read(p), read(s);
  checkIfImpossible();
  for (int i = 0; i <= 2; ++i) {
    memset(cnt, 0, sizeof(cnt));
    dfs(a[i], 1, r + p + s);
    if (cnt[0] == r && cnt[1] == p && cnt[2] == s) {
      return (void)printf("%s", ch + 1);
    }
  }
  puts("IMPOSSIBLE");
}

}  // namespace STDquantum
